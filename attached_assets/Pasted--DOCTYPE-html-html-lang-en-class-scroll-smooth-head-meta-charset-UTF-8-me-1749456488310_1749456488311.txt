    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Mobile menu toggle
        const mobileMenuBtn = document.getElementById('mobileMenuBtn');
        const mobileMenu = document.getElementById('mobileMenu');

        mobileMenuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        // Dark mode toggle
        const darkModeToggle = document.getElementById('darkModeToggle');
        const html = document.documentElement;

        // Check for saved theme preference or default to light mode
        const currentTheme = localStorage.getItem('theme') || 'light';
        if (currentTheme === 'dark') {
            html.classList.add('dark');
        }

        darkModeToggle.addEventListener('click', () => {
            html.classList.toggle('dark');
            const theme = html.classList.contains('dark') ? 'dark' : 'light';
            localStorage.setItem('theme', theme);
        });

        // Smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

        // Set minimum date for check-in and check-out
        const today = new Date().toISOString().split('T')[0];
        const dateInputs = document.querySelectorAll('input[type="date"]');
        dateInputs.forEach(input => {
            input.setAttribute('min', today);
        });

        // Update check-out minimum date when check-in changes
        const checkinInput = dateInputs[0];
        const checkoutInput = dateInputs[1];

        if (checkinInput && checkoutInput) {
            checkinInput.addEventListener('change', function() {
                const checkinDate = new Date(this.value);
                checkinDate.setDate(checkinDate.getDate() + 1);
                const minCheckout = checkinDate.toISOString().split('T')[0];
                checkoutInput.setAttribute('min', minCheckout);
                
                // Clear checkout if it's before the new minimum
                if (checkoutInput.value && checkoutInput.value <= this.value) {
                    checkoutInput.value = '';
                }
            });
        }

        // Add loading state to search button
        const searchButton = document.querySelector('button[type="submit"], button:has(i[data-lucide="search"])');
        if (searchButton) {
            searchButton.addEventListener('click', function(e) {
                e.preventDefault();
                this.disabled = true;
                this.innerHTML = '<i data-lucide="loader-2" class="w-5 h-5 animate-spin"></i><span>Searching...</span>';
                
                // Simulate search delay
                setTimeout(() => {
                    this.disabled = false;
                    this.innerHTML = '<i data-lucide="search" class="w-5 h-5"></i><span>Search Hotels</span>';
                    lucide.createIcons();
                }, 2000);
            });
        }

        // Add heart toggle functionality
        document.querySelectorAll('i[data-lucide="heart"]').forEach(heart => {
            heart.addEventListener('click', function(e) {
                e.stopPropagation();
                this.classList.toggle('fill-current');
                this.classList.toggle('text-red-500');
            });
        });

        // Filter functionality
        const filterInputs = document.querySelectorAll('input[type="checkbox"], input[type="range"]');
        filterInputs.forEach(input => {
            input.addEventListener('change', function() {
                // Here you would typically filter the hotel results
                console.log('Filter changed:', this.type, this.checked || this.value);
            });
        });

        // Clear filters functionality
        const clearFiltersBtn = document.querySelector('button:contains("Clear All Filters")');
        if (clearFiltersBtn) {
            clearFiltersBtn.addEventListener('click', function() {
                filterInputs.forEach(input => {
                    if (input.type === 'checkbox') {
                        input.checked = false;
                    } else if (input.type === 'range') {
                        input.value = input.getAttribute('max') / 2;
                    }
                });
            });
        }
    </script>
